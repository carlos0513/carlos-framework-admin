## 1. 规范总则

### 1.1 核心原则

- **一致性**: 与项目现有代码风格保持一致
- **可读性**: 代码是写给人看的，顺便给机器执行
- **可维护性**: 易于理解、修改和扩展
- **性能**: 在保证可读性的前提下考虑性能

### 1.2 强制等级

| 等级     | 标识 | 说明        |
|--------|----|-----------|
| **强制** | 🔴 | 必须遵守，不得违反 |
| **推荐** | 🟡 | 强烈建议遵守    |
| **参考** | 🟢 | 可根据实际情况选择 |

### 2.2 文件组织规则 🔴

**页面组件**:

```
views/manage/user/index.vue           # 用户列表页
```

- 复用组件提取到 `components` 目录

## 7. 状态管理规范

### 7.1 Store 组织 🔴

```typescript
// src/store/modules/user.ts
import { defineStore } from 'pinia';

interface UserState {
  userInfo: Api.Auth.UserInfo | null;
  token: string;
  permissions: string[];
}

export const useUserStore = defineStore('user', () => {
  // ========== State ==========
  const state = reactive<UserState>({
    userInfo: null,
    token: localStorage.getItem('token') || '',
    permissions: []
  });

  // ========== Getters ==========
  const isLogin = computed(() => !!state.token);
  const userName = computed(() => state.userInfo?.userName || '');

  // ========== Actions ==========
  async function login(loginData: LoginParams): Promise<void> {
    const { data } = await fetchLogin(loginData);
    state.token = data.token;
    localStorage.setItem('token', data.token);
  }

  function logout(): void {
    state.token = '';
    state.userInfo = null;
    localStorage.removeItem('token');
  }

  // ========== Return ==========
  return {
    ...toRefs(state),  // 展开响应式数据
    isLogin,
    userName,
    login,
    logout
  };
});
```

### 7.2 Store 使用规范 🔴

```vue
<script setup lang="ts">
// ✅ 解构时使用 storeToRefs 保持响应式
import { storeToRefs } from 'pinia';
import { useUserStore } from '@/store/modules/user';

const userStore = useUserStore();
const { userInfo, token } = storeToRefs(userStore);  // 保持响应式
const { login, logout } = userStore;  // 方法直接解构

// ❌ 错误：直接解构会丢失响应式
const { userInfo } = useUserStore();  // userInfo 不会更新
</script>
```

### 7.3 状态修改原则 🔴

```typescript
// ✅ 好的做法：在 Store 内封装修改逻辑
// store
function updateUserName(name: string) {
  state.userInfo!.userName = name;
}

// 组件
userStore.updateUserName('新名称');

// ❌ 避免：在组件中直接修改 Store 状态
userStore.userInfo!.userName = '新名称';  // 可以运行，但不推荐
```

## 8. API 请求规范

### 8.1 API 文件组织 🔴

```typescript
// src/service/api/user.ts
import { request } from '../request';

/**
 * 获取用户列表
 * @param params - 查询参数
 */
export function fetchGetUserList(params?: Api.SystemManage.UserSearchParams) {
  return request<Api.SystemManage.UserList>({
    url: '/systemManage/getUserList',
    method: 'get',
    params
  });
}

/**
 * 新增用户
 * @param data - 用户数据
 */
export function fetchAddUser(data: Api.SystemManage.UserCreateParams) {
  return request<boolean>({
    url: '/systemManage/addUser',
    method: 'post',
    data
  });
}

// 命名规范：fetch + 动词 + 资源名
// fetch: 表示发起请求
// 动词: Get/Add/Update/Delete
// 资源: User/Role/Menu
```

### 8.2 请求使用规范 🔴

```vue
<script setup lang="ts">
import { fetchGetUserList, fetchDeleteUser } from '@/service/api/user';

async function loadData() {
  loading.value = true;
  try {
    // ✅ 使用解构获取 data 和 error
    const { data, error } = await fetchGetUserList(searchParams);

    if (!error && data) {
      userList.value = data.records;
      pagination.total = data.total;
    }
  } finally {
    loading.value = false;
  }
}

async function handleDelete(id: number) {
  const { error } = await fetchDeleteUser(id);
  if (!error) {
    message.success('删除成功');
    loadData();
  }
}
</script>
```
